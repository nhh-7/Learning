  ```go
  func partition(nums []int, left, right int) int {
  	random_idx := left + rand.Intn(right-left+1)
  	nums[left], nums[random_idx] = nums[random_idx], nums[left]
  	pivot := nums[left]
  
  	i, j := left+1, right
  	for {
      // 寻找第一个大于或等于 pivot 的数。
  		for i <= j && nums[i] < pivot {
  			i++
  		}
      // 寻找第一个小于或等于 pivot 的数。
  		for i <= j && nums[j] > pivot {
  			j--
  		}
  		if i >= j {
  			break
  		}
  		nums[i], nums[j] = nums[j], nums[i]
  		i++
  		j--
  	}
    // 当 i 和 j 相遇并错开后，j 指向的位置正是最后一个小于等于 Pivot 的元素。将 Pivot 换到这里，Pivot 就到了它在有序序列中“绝对正确”的位置。
  	nums[left], nums[j] = nums[j], nums[left]
  	return j
  }
  ```

##### 背包问题

 0/1 背包模板 (物品有限)

```go
// weights: 物品重量, values: 物品价值, W: 背包总容量
func knapsack01(weights []int, values []int, W int) int {
    dp := make([]int, W+1) // 默认初始化为0
    
    // 遍历物品
    for i, w := range weights {
        v := values[i]
        // 遍历容量：【关键】必须倒序！保证物品只被放入一次
        for j := W; j >= w; j-- {
            dp[j] = max(dp[j], dp[j-w] + v)
        }
    }
    return dp[W]
}
```

完全背包模板（物品无限）

```go
func knapsackUnbounded(weights []int, values []int, W int) int {
    dp := make([]int, W+1)
    
    // 遍历物品
    for i, w := range weights {
        v := values[i]
        // 遍历容量：【关键】必须正序！因为当前物品可以被多次使用
        for j := w; j <= W; j++ {
            dp[j] = max(dp[j], dp[j-w] + v)
        }
    }
    return dp[W]
}
```

如果题目是求**“排列数”**还是**“组合数”**，循环顺序会有变化（常见于完全背包）：

1. **求组合数**（顺序无关，如 `[1, 2]` 和 `[2, 1]` 视为同一种）：
   - **先遍历物品，再遍历背包**，固定物品顺序，避免重复计算排列。
   - 典型题目：LeetCode 518 零钱兑换 II。
2. **求排列数**（顺序有关，如 `[1, 2]` 和 `[2, 1]` 是不同的）：
   - **先遍历背包，再遍历物品**，每一位容量都尝试所有物品，考虑顺序。
   - 典型题目：LeetCode 377 组合总和 Ⅳ。
